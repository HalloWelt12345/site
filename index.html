let tamagotchi = { hunger: 50, happiness: 50, energy: 50, sleep: false, awake: false, eating: false, playing: false, age: 0 }; const tamagotchiElement = document.getElementById("tamagotchi"); function updateTamagotchi() { if (tamagotchi.hunger < 0) { tamagotchi.hunger = 100; } if (tamagotchi.happiness < 0) { tamagotchi.happiness = 100; } if (tamagotchi.energy < 0) { tamagotchi.energy = 100; } if (tamagotchi.hunger > 100) { tamagotchi.hunger = 0; } if (tamagotchi.happiness > 100) { tamagotchi.happiness = 0; } if (tamagotchi.energy > 100) { tamagotchi.energy = 0; } tamagotchiElement.innerHTML = `

Hunger: ${tamagotchi.hunger}

Happiness: ${tamagotchi.happiness}

Energy: ${tamagotchi.energy}

Age: ${tamagotchi.age}

`; } function feedTamagotchi() { tamagotchi.hunger -= 10; tamagotchi.energy -= 5; tamagotchi.happiness += 10; updateTamagotchi(); } function playWithTamagotchi() { tamagotchi.happiness -= 10; tamagotchi.energy -= 10; tamagotchi.hunger += 5; updateTamagotchi(); } function sleepTamagotchi() { tamagotchi.energy += 20; tamagotchi.happiness -= 5; tamagotchi.hunger += 5; updateTamagotchi(); } setInterval(() => { tamagotchi.age += 1; tamagotchi.hunger += 1; tamagotchi.happiness -= 1; tamagotchi.energy -= 1; updateTamagotchi(); }, 1000);
